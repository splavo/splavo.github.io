AI Cover Letter Writer

import openai openai.api_key = "XX" name = input("Full Name: ") skills = input("Skills: ") experience = input("Experience: ") position = input("Position Applying For: ") personinputs = { "name": name, "skills": skills, "experience": experience, "position": position } response = openai.Completion.create( model="text-davinci-002", prompt="create a cover letter for {name}. They are applying for the position of {position}. Their skills are: {skills}. They previously worked as a: {experience}.".format(name=personinputs['name'], skills=personinputs['skills'], experience=personinputs['experience'], position=personinputs['position']), temperature=0, max_tokens=200, top_p=1, frequency_penalty=0, presence_penalty=0 ) print(response.choices[0].text)
